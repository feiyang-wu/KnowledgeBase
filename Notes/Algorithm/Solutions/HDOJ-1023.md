# HDOJ-1023 Train Problem II
## 题目大意
给定一个入栈序列，一个出栈序列，求是否可行
## 解题思路
栈+模拟
## 细节问题
无
## 代码实现
```
// Author: Feiyang Wu

// Space for include
#include <iostream>
#include <string>
#include <map>
#include <cmath>
#include <vector>
#include <algorithm>

// Space for using
using namespace std;

// Space for define
#define MAXN 1000000

// Space for vars
int ans[20];// 0 for in, 1 for out
int stack[10];
bool vis[10];//

// Space for funcs
inline int char_to_int(char c) { return c - '0'; }

int main()
{
	int n;
	string strin, strout;
		
	while (cin >> n >> strin >> strout)
	{
		bool flag = true;
		int top = -1;
		int cntin = 0;
		int cntans = 0;
		for (int i = 1; i <= n; i++) vis[i] = false;
		for (int i = 0;i < n; i++)
		{
			while (vis[char_to_int(strout[i])] == false)
			{
				top++;
				stack[top] = char_to_int(strin[cntin]);
				vis[char_to_int(strin[cntin])] = true;
				ans[cntans] = 0;
				cntans++;
				cntin++;
			}
			if (stack[top] == char_to_int(strout[i]))
			{
				top--;
				ans[cntans] = 1;
				cntans++;
			}
			else
			{
				flag = false;
				break;
			}
		}
		if (flag)
		{
			cout << "Yes." << endl;
			for (int i = 0; i < cntans; i++)
			{
				if(ans[i]) cout << "out" << endl;
				else cout << "in" << endl;
			}
			cout << "FINISH" << endl;
		}
		else
		{
			cout << "No." << endl << "FINISH" << endl;
		}
	}
	return 0;
}
```

# Tempter of the Bone
## 题目大意
走迷宫，走过的地方不能再走，墙不能走，问T秒时，主人公能不能正好从门出去。

图例：
```
'X': a block of wall, which the doggie cannot enter; 
'S': the start point of the doggie; 
'D': the Door; or
'.': an empty block.

S.X.
..X.
..XD
....
```
## 解题思路
带visited数组标记的DFS
## 细节问题
写代码要仔细看，WA了两发只是因为把`y-1`写成`y+1`了。
## 代码实现
```
#include <iostream>
#include <string>
#include <map>
#include <cmath>
#include <vector>
#include <algorithm>


using namespace std;

int __map[7][7];// 0 for door, 1 for road, 2 for wall, 3 for start;
bool visited[7][7];
bool success;
int m, n, t;

bool run(int x, int y)
{
	if (x < 0 || y < 0 || x >= m || y >= n) return false;
	if (visited[x][y]) return false;
	if (t == 0 && __map[x][y] == 0) return true;
	if (__map[x][y] == 2 || __map[x][y] == 0) return false;
	

	t--; visited[x][y] = true;
	
	bool temp = run(x - 1, y);
	if (temp) return true;
	temp = run(x, y - 1);
	if (temp) return true;
	temp = run(x + 1, y);
	if (temp) return true;
	temp = run(x, y + 1);
	if (temp) return true;

	t++; visited[x][y] = false;
	return false;
}

int main()
{
	while (scanf("%d%d%d", &n, &m, &t) && m != 0)
	{
		int start[2];
		for (int j = 0; j < n; j++)
		{
			for (int i = 0; i < m; i++)
			{
				char s;
				cin >> s;
				if (s == 'S')
				{
					start[0] = i;
					start[1] = j;
					__map[i][j] = 3;
					visited[i][j] = false;
				}
				else if (s == 'X')
				{
					__map[i][j] = 2;
					visited[i][j] = false;
				}
				else if (s == '.')
				{
					__map[i][j] = 1;
					visited[i][j] = false;
				}
				else if (s == 'D')
				{
					__map[i][j] = 0;
					visited[i][j] = false;
				}
			}
		}
		success = false;
		if (run(start[0], start[1])) printf("YES\n");
		else printf("NO\n");

	}
	return 0;
}
```

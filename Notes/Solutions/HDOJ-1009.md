# HD0J-1009 FatMouse' Trade
## 题目大意
杰瑞拿猫粮跟汤姆换吃的，贪心
## 解题思路
按性价比sort一下，完事
## 细节问题
无
## 代码实现
```
#include <iostream>
#include <string>
#include <map>
#include <cmath>
#include <vector>
#include <algorithm>

#define MAXN 1005

using namespace std;

struct food {
	int weight;
	int cost;
	double ratio;
};

food warehouse[MAXN];

bool compareRatio(food a, food b)
{
	return a.ratio > b.ratio;
}

int main()
{
	int m, n;
	while (scanf("%d%d", &m, &n) && m != -1)
	{
		for (int i = 0; i < n; i++)
		{

			scanf("%d%d", &warehouse[i].weight, &warehouse[i].cost);
			warehouse[i].ratio = (double)warehouse[i].weight / (double)warehouse[i].cost;
		}
		sort(warehouse, warehouse + n, compareRatio);
		double remain = m;
		double sum = 0;
		for (int i = 0; i < n; i++)
		{
			if (remain == 0.f) break;
			else if (remain > warehouse[i].cost)
			{
				sum += warehouse[i].weight;
				remain -= warehouse[i].cost;
			}
			else
			{
				sum += (remain / warehouse[i].cost*warehouse[i].weight);
				break;
			}
		}
		printf("%.3lf\n", sum);

	}
	return 0;
}
```
